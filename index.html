Palindrome


Method 1:
function palindrome(str) {
var str = str.toLowerCase().replace(' ', '');
var palin = str.split("").reverse().join("");
if (palin === str){
return true;
} else {
return false;
}
}


Method 2:
function palindrome(passage) {
// normalize to lower case.
passage = passage.toLowerCase();
// get rid of spaces.
passage = passage.replace(' ', '');
// you could reverse it if you want to here.
for (var i = 0; i < passage.length; i++) {
if (passage[i] !== passage[(passage.length -1) - i]) {
return false;
}
}
return true;
}

FizzBuzz


function FizzBuzz() {
let num = [];
for (let i = 1; i < 101; i++) {
if (i % 3 === 0 && i % 5 === 0) {
num.push('FizzBuzz');
} else if (i % 5 === 0) {
num.push('Buzz');
} else if (i % 3 === 0) {
num.push('Fizz');
} else {
num.push(i);
}
}
console.log(num);
};
